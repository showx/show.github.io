   有时候程序要批量处理指定时间范围的东西，使用此函数： 
   /**
     * 生成指定范围日期
     * @var string
     */
    public static createday($d0 = '2010-02-21',$d1 = '2010-04-23'; )
    {
        $_time = range(strtotime($d0), strtotime($d1), 24*60*60);
        $_time = array_map(create_function('$v', 'return date("Y-m-d", $v);'), $_time);
        return $_time;
    }
