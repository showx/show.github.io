&lt;script type="text/javascript"&gt;
(function(){
var wait = document.getElementById('wait'),href = document.getElementById('href').href;
var interval = setInterval(function(){
var time = --wait.innerHTML;
if(time &lt;= 0) {
location.href = href;
clearInterval(interval);
};
}, 1000);
})();
&lt;/script&gt;
